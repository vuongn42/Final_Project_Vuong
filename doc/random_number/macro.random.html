Generate a random number.

Examples

let f: f64 = random_number::random!();

assert!(0.0 <= f && f <= 1.0);

let i: u8 = random_number::random!(..);

assert!(0 <= i && i <= 255);
use random_number::rand;

let f: f64 = random_number::random!(thread_rng);

assert!(0.0 <= f && f <= 1.0);

assert!(0 <= i && i <= 255);
